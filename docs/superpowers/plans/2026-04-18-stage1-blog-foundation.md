# Stage 1 Blog Foundation Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the first complete Raptor blog milestone with a homepage, article page, archive page, tag pages, category pages, and shared navigation before any visual recreation work.

**Architecture:** Keep the implementation Raptor-native and Swift-first. Use custom static `Page` types for archive, tags, and categories so route names can match `/archives`, `/tags`, and `/categories` exactly, while using a small pure-Swift indexing layer to discover tags and category folders from the `Posts/` tree. Use one shared layout and a handful of components so Stage 2 styling can happen without reshaping routes or data flow.

**Tech Stack:** Swift 6.2, Raptor 0.1.2, SwiftPM tests, Markdown posts with YAML front matter

## Route Plan

- `/`
  Homepage
- `/<post-slug>`
  Post detail routes generated by Raptor from `Posts/`
- `/archives`
  Archive index page
- `/tags`
  Tag index page
- `/tags/<slug>`
  Tag detail pages
- `/categories`
  Category index page
- `/categories/<slug>`
  Category detail pages

## Data Model Decisions

- **Tags:** Read from YAML front matter `tags:` and support comma-separated values, matching Raptor’s own post metadata convention.
- **Categories:** For Stage 1, derive category from the first subfolder under `Posts/`, e.g. `Posts/devlog/file.md` becomes category `devlog`. This keeps the implementation Swifty, avoids inventing a second metadata parser path, and aligns with Raptor’s built-in `post.type`.
- **Archive buckets:** Group by `yyyy-MM` using `Post.date`.
- **Generated page registry:** Build tag/category page instances in `Site.prepare()` from the discovered source documents, then expose them through `pages`.

Create `Sources/Support/BlogContentDiscovery.swift`:

```swift
import Foundation

struct BlogSourceDocument: Equatable, Sendable {
    let relativePath: String
    let categoryFolder: String?
    let tags: [String]
}

enum BlogContentDiscovery {
    static func parse(relativePath: String, markdown: String) -> BlogSourceDocument {
        let categoryFolder = categoryFolder(from: relativePath)
        let tags = parseTags(from: markdown)
        return BlogSourceDocument(
            relativePath: relativePath,
            categoryFolder: categoryFolder,
            tags: tags
        )
    }

    static func tagEntries(from documents: [BlogSourceDocument]) -> [BlogTagEntry] {
        let counts = documents
            .flatMap(\.tags)
            .reduce(into: [String: Int]()) { result, tag in
                result[tag, default: 0] += 1
            }

        return counts
            .map { name, count in
                BlogTagEntry(
                    name: name,
                    slug: slugify(name),
                    count: count
                )
            }
            .sorted { lhs, rhs in
                lhs.name.localizedCaseInsensitiveCompare(rhs.name) == .orderedAscending
            }
    }

    static func categoryEntries(from documents: [BlogSourceDocument]) -> [BlogCategoryEntry] {
        let counts = documents
            .compactMap(\.categoryFolder)
            .reduce(into: [String: Int]()) { result, folder in
                result[folder, default: 0] += 1
            }

        return counts
            .map { folder, count in
                BlogCategoryEntry(
                    name: displayName(fromFolder: folder),
                    slug: slugify(folder),
                    sourceFolder: folder,
                    count: count
                )
            }
            .sorted { lhs, rhs in
                lhs.name.localizedCaseInsensitiveCompare(rhs.name) == .orderedAscending
            }
    }

    static func parseTags(from markdown: String) -> [String] {
        let frontMatter = frontMatterBlock(in: markdown)
        let lines = frontMatter.split(separator: "\n").map(String.init)

        guard let rawValue = lines.first(where: { $0.lowercased().hasPrefix("tags:") }) else {
            return []
        }

        let tagString = rawValue
            .split(separator: ":", maxSplits: 1)
            .dropFirst()
            .first?
            .trimmingCharacters(in: .whitespacesAndNewlines) ?? ""

        guard !tagString.isEmpty else { return [] }

        return tagString
            .split(separator: ",")
            .map { $0.trimmingCharacters(in: .whitespacesAndNewlines) }
            .filter { !$0.isEmpty }
    }

    static func categoryFolder(from relativePath: String) -> String? {
        let trimmed = relativePath.trimmingCharacters(in: CharacterSet(charactersIn: "/"))
        let components = trimmed.split(separator: "/").map(String.init)

        guard components.count >= 2 else { return nil }
        return components.first
    }

    static func slugify(_ text: String) -> String {
        text
            .lowercased()
            .replacingOccurrences(of: " ", with: "-")
    }

    static func displayName(fromFolder folder: String) -> String {
        folder
            .split(separator: "-")
            .map { $0.prefix(1).uppercased() + $0.dropFirst() }
            .joined(separator: " ")
    }

    private static func frontMatterBlock(in markdown: String) -> String {
        let lines = markdown.split(separator: "\n", omittingEmptySubsequences: false).map(String.init)

        guard lines.first == "---" else { return "" }

        var collected: [String] = []
        for line in lines.dropFirst() {
            if line == "---" {
                break
            }
            collected.append(line)
        }

        return collected.joined(separator: "\n")
    }
}
```

Create `Sources/Support/ArchiveSections.swift`:

```swift
import Foundation
import Raptor

enum ArchiveSections {
    static func make(from posts: [Post]) -> [ArchiveSection] {
        let formatter = DateFormatter()
        formatter.calendar = Calendar(identifier: .gregorian)
        formatter.locale = Locale(identifier: "en_US_POSIX")
        formatter.dateFormat = "yyyy-MM"

        let grouped = Dictionary(grouping: posts) { post in
            formatter.string(from: post.date)
        }

        return grouped
            .map { key, groupedPosts in
                ArchiveSection(
                    id: key,
                    title: key,
                    posts: groupedPosts.sorted { $0.date > $1.date }
                )
            }
            .sorted { $0.id > $1.id }
    }
}
```

import XCTest
@testable import RaptorArknights
import Raptor

final class BlogContentDiscoveryTests: XCTestCase {
    func testParseTagsReadsCommaSeparatedFrontMatter() {
        let markdown = """
        ---
        title: Sample
        tags: swift, raptor, arknights
        ---
        Hello
        """

        let document = BlogContentDiscovery.parse(
            relativePath: "notes/welcome.md",
            markdown: markdown
        )

        XCTAssertEqual(document.categoryFolder, "notes")
        XCTAssertEqual(document.tags, ["swift", "raptor", "arknights"])
    }

    func testTagEntriesAggregateCounts() {
        let documents = [
            BlogSourceDocument(relativePath: "notes/a.md", categoryFolder: "notes", tags: ["swift", "raptor"]),
            BlogSourceDocument(relativePath: "devlog/b.md", categoryFolder: "devlog", tags: ["raptor"])
        ]

        XCTAssertEqual(
            BlogContentDiscovery.tagEntries(from: documents),
            [
                BlogTagEntry(name: "raptor", slug: "raptor", count: 2),
                BlogTagEntry(name: "swift", slug: "swift", count: 1)
            ]
        )
    }

    func testCategoryEntriesUseFirstFolderUnderPosts() {
        let documents = [
            BlogSourceDocument(relativePath: "notes/a.md", categoryFolder: "notes", tags: []),
            BlogSourceDocument(relativePath: "notes/b.md", categoryFolder: "notes", tags: []),
            BlogSourceDocument(relativePath: "devlog/c.md", categoryFolder: "devlog", tags: [])
        ]

        XCTAssertEqual(
            BlogContentDiscovery.categoryEntries(from: documents),
            [
                BlogCategoryEntry(name: "Devlog", slug: "devlog", sourceFolder: "devlog", count: 1),
                BlogCategoryEntry(name: "Notes", slug: "notes", sourceFolder: "notes", count: 2)
            ]
        )
    }

    func testArchiveSectionsGroupPostsByYearMonth() {
        var april = Post()
        april.title = "April"
        april.metadata["date"] = ISO8601DateFormatter().date(from: "2026-04-18T00:00:00Z")
        april.metadata["type"] = "notes"

        var march = Post()
        march.title = "March"
        march.metadata["date"] = ISO8601DateFormatter().date(from: "2026-03-01T00:00:00Z")
        march.metadata["type"] = "notes"

        let sections = ArchiveSections.make(from: [march, april])

        XCTAssertEqual(sections.map(\.title), ["2026-04", "2026-03"])
        XCTAssertEqual(sections.first?.posts.first?.title, "April")
    }
}

@main
struct RaptorWebsite {

        }
    }

struct Arknights: Site {
    var name = "My Blog"
    var titleSuffix = " – My Awesome Site"
    var url = URL(static: "https://www.shiinayane.com")
    var author = "shiinayane"

    var discoveredTags: [BlogTagEntry] = []
    var discoveredCategories: [BlogCategoryEntry] = []

    var homePage = Home()
    var layout = MainLayout()

    var pages: [any Page] {
        [ArchivePage(), TagIndexPage(tags: discoveredTags), CategoryIndexPage(categories: discoveredCategories)]
        + discoveredTags.map(TagDetailPage.init)
        + discoveredCategories.map(CategoryDetailPage.init)
    }

    var postPages: [any PostPage] {
        ArticlePage()
    }

    mutating func prepare() async throws {
        let postsDirectory = URL(fileURLWithPath: FileManager.default.currentDirectoryPath)
            .appending(path: "Posts")

        let markdownFiles = FileManager.default.enumerator(
            at: postsDirectory,
            includingPropertiesForKeys: nil
        )?
        .compactMap { $0 as? URL }
        .filter { $0.pathExtension == "md" } ?? []

        let documents = try markdownFiles.map { fileURL in
            let markdown = try String(contentsOf: fileURL, encoding: .utf8)
            let relativePath = fileURL.path.replacingOccurrences(of: postsDirectory.path + "/", with: "")
            return BlogContentDiscovery.parse(relativePath: relativePath, markdown: markdown)
        }

        discoveredTags = BlogContentDiscovery.tagEntries(from: documents)
        discoveredCategories = BlogContentDiscovery.categoryEntries(from: documents)
    }
}

import Raptor

struct ArticlePage: PostPage {
    @Environment(\.posts) private var posts

    var body: some HTML {
        VStack(alignment: .leading) {
            Text(post.title)
                .font(.title1)

            if let subtitle = post.subtitle {
                Text(subtitle)
                    .foregroundStyle(.secondary)
                    .margin(.bottom, 12)
            }

            PostMeta(post: post)
                .margin(.bottom, 24)

            post.text

            HStack {
                if let previousPost {
                    LinkGroup(destination: previousPost) {
                        Text("Previous")
                        Text(previousPost.title)
                    }
                }

                Spacer()

                if let nextPost {
                    LinkGroup(destination: nextPost) {
                        Text("Next")
                        Text(nextPost.title)
                    }
                }
            }
            .margin(.top, 40)
        }
    }

    private var currentPostIndex: Int? {
        posts.firstIndex { $0.path == post.path }
    }

    private var previousPost: Post? {
        guard let currentPostIndex, currentPostIndex > 0 else { return nil }
        return posts[currentPostIndex - 1]
    }

    private var nextPost: Post? {
        guard let currentPostIndex, currentPostIndex < posts.count - 1 else { return nil }
        return posts[currentPostIndex + 1]
    }
}

struct TagDetailPage: Page {
    let tag: BlogTagEntry

    var title: String { "Tag: \(tag.name)" }
    var path: String { tag.path }

    @Environment(\.posts) private var posts

    var body: some HTML {
        let matchingPosts = posts.filter { post in
            post.tags?.contains(where: { $0.name == tag.name }) == true
        }

        VStack(alignment: .leading) {
            Text(tag.name)
                .font(.title1)
                .margin(.bottom, 20)

            List {
                ForEach(matchingPosts) { post in
                    PostListItem(post: post)
                }
            }
